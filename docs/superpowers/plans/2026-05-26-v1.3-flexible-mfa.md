Existing `MFAEnroll` and `MFAChallenge` are refactored from "TOTP only" into thin dispatchers that branch on user-picked factor type. Lockout prevention by client-side guard requiring ≥1 verified factor at all times.

**Tech Stack:** React 18, TypeScript (strict), `@supabase/supabase-js` v2, browser-native `navigator.credentials.create/get`, Vitest + RTL.

```ts
import { supabase } from '../lib/supabase';

export type FactorType = 'totp' | 'webauthn';

export type EnrolledFactor = {
  id: string;
  type: FactorType;
  friendlyName: string;
  createdAt: string;
};

export class LastFactorError extends Error {
  constructor() {
    super('Cannot remove the last verified MFA factor.');
    this.name = 'LastFactorError';
  }
}

export async function listFactors(): Promise<EnrolledFactor[]> {
  const { data, error } = await supabase.auth.mfa.listFactors();
  if (error) throw error;
  const all: EnrolledFactor[] = [];
  for (const f of data.totp) {
    if (f.status !== 'verified') continue;
    all.push({
      id: f.id,
      type: 'totp',
      friendlyName: f.friendly_name ?? 'Authenticator',
      createdAt: f.created_at,
    });
  }
  // Supabase's listFactors also returns phone/webauthn arrays in newer SDK versions.
  // Treat unknown shape defensively.
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const dataAny = data as any;
  if (Array.isArray(dataAny.webauthn)) {
    for (const f of dataAny.webauthn) {
      if (f.status !== 'verified') continue;
      all.push({
        id: f.id,
        type: 'webauthn',
        friendlyName: f.friendly_name ?? 'Passkey',
        createdAt: f.created_at,
      });
    }
  }
  return all;
}

export async function enrollTotp(friendlyName: string): Promise<{
  factorId: string;
  qrCode: string;
  secret: string;
}> {
  const { data, error } = await supabase.auth.mfa.enroll({
    factorType: 'totp',
    friendlyName,
  });
  if (error) throw error;
  return {
    factorId: data.id,
    qrCode: data.totp.qr_code,
    secret: data.totp.secret,
  };
}

export async function verifyTotpEnrollment(args: {
  factorId: string;
  code: string;
}): Promise<void> {
  const challenge = await supabase.auth.mfa.challenge({ factorId: args.factorId });
  if (challenge.error) throw challenge.error;
  const verify = await supabase.auth.mfa.verify({
    factorId: args.factorId,
    challengeId: challenge.data.id,
    code: args.code,
  });
  if (verify.error) throw verify.error;
}

type PasskeyEnrollResult = {
  factorId: string;
  credentialCreationOptions: PublicKeyCredentialCreationOptions;
};

export async function enrollPasskey(friendlyName: string): Promise<PasskeyEnrollResult> {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const { data, error } = await (supabase.auth.mfa.enroll as any)({
    factorType: 'webauthn',
    friendlyName,
  });
  if (error) throw error;
  // Supabase returns the WebAuthn options in `data.webauthn.credential_creation_options`
  // or a similarly shaped field depending on SDK version. Map to the standard browser type.
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const dataAny = data as any;
  const opts =
    dataAny.webauthn?.credential_creation_options ??
    dataAny.webauthn?.credentialCreationOptions;
  if (!opts) throw new Error('Supabase did not return credential creation options.');
  return {
    factorId: data.id,
    credentialCreationOptions: opts as PublicKeyCredentialCreationOptions,
  };
}

export async function verifyPasskeyEnrollment(args: {
  factorId: string;
  credential: PublicKeyCredential;
}): Promise<void> {
  const challenge = await supabase.auth.mfa.challenge({ factorId: args.factorId });
  if (challenge.error) throw challenge.error;
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const verify = await (supabase.auth.mfa.verify as any)({
    factorId: args.factorId,
    challengeId: challenge.data.id,
    credential: args.credential,
  });
  if (verify.error) throw verify.error;
}

export async function challengeTotp(factorId: string): Promise<{ challengeId: string }> {
  const { data, error } = await supabase.auth.mfa.challenge({ factorId });
  if (error) throw error;
  return { challengeId: data.id };
}

export async function verifyTotpChallenge(args: {
  factorId: string;
  challengeId: string;
  code: string;
}): Promise<void> {
  const { error } = await supabase.auth.mfa.verify({
    factorId: args.factorId,
    challengeId: args.challengeId,
    code: args.code,
  });
  if (error) throw error;
}

type PasskeyChallengeResult = {
  challengeId: string;
  credentialRequestOptions: PublicKeyCredentialRequestOptions;
};

export async function challengePasskey(factorId: string): Promise<PasskeyChallengeResult> {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const { data, error } = await (supabase.auth.mfa.challenge as any)({ factorId });
  if (error) throw error;
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const dataAny = data as any;
  const opts =
    dataAny.webauthn?.credential_request_options ??
    dataAny.webauthn?.credentialRequestOptions;
  if (!opts) throw new Error('Supabase did not return credential request options.');
  return {
    challengeId: data.id,
    credentialRequestOptions: opts as PublicKeyCredentialRequestOptions,
  };
}

export async function verifyPasskeyChallenge(args: {
  factorId: string;
  challengeId: string;
  credential: PublicKeyCredential;
}): Promise<void> {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const { error } = await (supabase.auth.mfa.verify as any)({
    factorId: args.factorId,
    challengeId: args.challengeId,
    credential: args.credential,
  });
  if (error) throw error;
}

export async function unenrollFactor(factorId: string): Promise<void> {
  const current = await listFactors();
  if (current.length <= 1 && current.some((f) => f.id === factorId)) {
    throw new LastFactorError();
  }
  const { error } = await supabase.auth.mfa.unenroll({ factorId });
  if (error) throw error;
}

import { useCallback, useEffect, useState } from 'react';
import { listFactors } from '../api/factors';
import type { EnrolledFactor } from '../api/factors';

export type UseFactorsResult = {
  factors: EnrolledFactor[];
  loading: boolean;
  error: string | null;
  refresh: () => Promise<void>;
};

export function useFactors(): UseFactorsResult {
  const [factors, setFactors] = useState<EnrolledFactor[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refresh = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const next = await listFactors();
      setFactors(next);
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    void refresh();
  }, [refresh]);

  return { factors, loading, error, refresh };
}
```

              key={type}
              type="button"

              </div>
              <div className="text-xs opacity-70 mt-0.5">{meta.helper}</div>

import { useEffect, useState } from 'react';
import type { FormEvent } from 'react';
import { enrollTotp, verifyTotpEnrollment } from '../api/factors';

type Props = {
  onDone: () => void;
  onCancel: () => void;
};

export default function TotpEnroll({ onDone, onCancel }: Props) {
  const [qr, setQr] = useState<string | null>(null);
  const [secret, setSecret] = useState<string | null>(null);
  const [factorId, setFactorId] = useState<string | null>(null);
  const [code, setCode] = useState('');
  const [error, setError] = useState<string | null>(null);
  const [busy, setBusy] = useState(false);

  useEffect(() => {
    let cancelled = false;
    async function start() {
      setBusy(true);
      try {
        const friendlyName = `Authenticator ${new Date().toISOString().slice(0, 10)}`;
        const result = await enrollTotp(friendlyName);
        if (cancelled) return;
        setFactorId(result.factorId);
        setQr(result.qrCode);
        setSecret(result.secret);
      } catch (e) {
        if (!cancelled) setError(e instanceof Error ? e.message : String(e));
      } finally {
        if (!cancelled) setBusy(false);
      }
    }
    void start();
    return () => { cancelled = true; };
  }, []);

  async function verify(e: FormEvent) {
    e.preventDefault();
    if (!factorId) return;
    setError(null);
    setBusy(true);
    try {
      await verifyTotpEnrollment({ factorId, code });
      onDone();
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className="mx-auto max-w-sm p-8 space-y-4">
      <div className="text-center">
        <div className="text-4xl mb-1">📱</div>
        <h1 className="text-xl font-extrabold">Add an authenticator app</h1>
      </div>
      <p className="text-sm text-muted">
        Scan this QR code with Google Authenticator, 1Password, Authy, etc., then enter the
        6-digit code to confirm.
      </p>
      {qr && (
        <img
          src={qr}
          alt="TOTP QR code"
          className="border border-highlight rounded-lg mx-auto bg-card p-2"
          width={200}
          height={200}
        />
      )}
      {secret && (
        <p className="text-xs text-muted break-all">
          Or enter this secret manually:{' '}
          <code className="bg-bg px-1 rounded">{secret}</code>
        </p>
      )}
      <form onSubmit={verify} className="space-y-2">
        <input
          type="text"
          inputMode="numeric"
          pattern="[0-9]{6}"
          maxLength={6}
          placeholder="123456"
          value={code}
          onChange={(e) => setCode(e.target.value)}
          className="w-full border border-highlight bg-card rounded-lg px-3 py-2 text-center text-lg tracking-widest"
          required

        {error && <p className="text-negative text-sm">{error}</p>}

          disabled={busy || code.length !== 6}
          className="w-full bg-hero-bg text-hero-text rounded-lg py-2 font-bold disabled:opacity-50"

          {busy ? 'Verifying…' : 'Verify and continue'}

        <button
          type="button"
          onClick={onCancel}
          className="w-full text-sm text-muted underline"

          Back
        </button>

    </div>
  );
}

    });
  } catch {
    return iso;
  }
}

const ICONS: Record<FactorType, string> = { webauthn: '🔐', totp: '📱' };
const LABELS: Record<FactorType, string> = { webauthn: 'Passkey', totp: 'Authenticator app' };

    items: items.map((i) => ({

      category(2, 'Loans', [{ id: 3, projected: 200, actual: 200 }]),
    ];

    expect(screen.getByText('$2.00')).toBeInTheDocument();    // diff +2 (over)
  });
